#print unique elements
a = [1, 2, 3, 6, 3, 4, 2]
ai = []
b=0
for i in a:
    b=b+1
for i in range(len(a)):
    is_duplicate = False
    for j in range(len(a)):
        if i != j and a[i] == a[j]:
            is_duplicate = True
            break

    if not is_duplicate and a[i] not in ai:
        ai.append(a[i])

print(ai)
